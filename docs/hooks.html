Hooks

React hooks for auth state, SDK access, and context management

useCalimero

Primary hook — returns the SDK instance and current auth state:

import { useCalimero } from '@calimero-network/mero-react';

const {
  sdk,             // MeroJs instance
  isAuthenticated,// boolean
  isLoading,      // boolean — true during init
  login,          // (username, password) => Promise<void>
  logout,         // () => void
} = useCalimero();

useCalimeroAdmin

Convenience hook for admin API operations:

import { useCalimeroAdmin } from '@calimero-network/mero-react';

function ContextList() {
  const { getContexts, createContext } = useCalimeroAdmin();

  const [contexts, setContexts] = useState([]);

  useEffect(() => {
    getContexts().then(setContexts);
  }, []);

  return <ul>{contexts.map(c => <li>{c.id}</li>)}</ul>;
}

useCalimeroRpc

Execute WASM methods directly from components:

import { useCalimeroRpc } from '@calimero-network/mero-react';

function PostList({ contextId }) {
  const { execute, query, mutate } = useCalimeroRpc(contextId);

  const loadPosts = async () => {
    return await query('get_posts', {});
  };

  const createPost = async (title) => {
    await mutate('create_post', { title });
  };
}

useCalimeroSubscription

Subscribe to real-time node events in a component:

import { useCalimeroSubscription } from '@calimero-network/mero-react';

function EventFeed({ contextId }) {
  const events = useCalimeroSubscription(contextId, {
    protocol: 'sse', // or 'ws'
  });

  return (
    <ul>
      {events.map(e => (
        <li>{e.eventType}: {JSON.stringify(e.payload)}</li>
      ))}
    </ul>
  );
}
